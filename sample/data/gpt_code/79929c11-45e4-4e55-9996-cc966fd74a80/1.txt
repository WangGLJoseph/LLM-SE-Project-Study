template <typename T>
void EntitySetStore<T>::insert(const T& element) {
  set.insert(element);
}

template <typename T>
std::unordered_set<T> EntitySetStore<T>::getAll() {
  return set;
}

template <typename T>
void EntitySetStore<T>::clear() {
  set.clear();
}

template class EntitySetStore<int>;
template class EntitySetStore<std::string>;
