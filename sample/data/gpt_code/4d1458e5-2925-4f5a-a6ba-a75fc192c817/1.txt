void addRow(std::shared_ptr<IDataEntry> dataEntry) override;

    std::shared_ptr<Value> getRow(Key key) {
        auto it = table.find(key);
        if (it != table.end()) {
            return it->second;
        }
        return nullptr;
    }

    std::unordered_map<Key, std::shared_ptr<Value>> getAllRows() {
        return table;
    }
